   logout(userID,isSocketId,callback){
       
       const data = {
             $set :{
                 online : 'N'
             }
         };
       this.Mongodb.onConnect( (db,ObjectID) => {
           
           let condition = {};
           if (isSocketId) {
               condition.socketId = userID;
           }else{
               condition._id = ObjectID(userID);
           }
 
 
           db.collection('users').update( condition, data ,(err, result) => {
               db.close();
               callback(err,result.result);
           });
       });
   }
